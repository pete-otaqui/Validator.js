  6     /**
  7      *  Simple Validator class, which can do quick one-off validations, or construct more complex multiple-rule validators.
  8      *  @name Validator
  9      *  @class
 10      *  @constructor
 11      *  @namespace
 12      *  @example
 13      *      alert( Validator.unique([1, 2, 3]) ); // true
 14      *      alert( Validator.unique([1, 2, 1]) ); // false
 15      *      alert( Validator.minLength([1, 2, 1], 3) ); // true
 16      *      
 17      *      myValidatorWillPass.add('unique')
 18      *      myValidatorWillPass.add('minLength', 3);
 19      *      myValidatorWillPass.validate( [1, 2, 3] ); // true, unique and long enough
 20      *      
 21      *      myValidatorWillFail.add('unique')
 22      *      myValidatorWillFail.add('minLength', 5);
 23      *      myValidatorWillFail.validate( [1, 2, 3] ); // false, unique but not long enough
 24      */
 25     Validator = function() {
 26         var myVs = {};
 27         return {
 28             errors : [],
 29             /**
 30              *  Adds a validation, supply the name as the first argument and all subsequent arguments will be passed on
 31              *  @name Validator#add
 32              *  @function
 33              *  @public
 34              *  @param {String} v the name of the validation to add
 35              *  @example
 36              *      var v = new Validator();
 37              *      v.add('minLength', 5); // create a validator that will check for minLength of 5
 38              *      v.validate( myArray ); // returns true / false
 39              */
 40             add: function(v) {
 41                 if ( typeof Validator[v] === 'undefined' ) {
 42                     throw new Error('unknown validation type: '+v);
 43                 }
 44                 if ( typeof myVs[v] === 'undefined' ) {
 45                     myVs[v] = [];
 46                 }
 47                 myVs[v].push(Array.prototype.slice.call(arguments, 1));
 48             },
 49             /**
 50              *  Runs the added validations agains the supplied content
 51              *  @name Validator#validate
 52              *  @function
 53              *  @public
 54              *  @param {String|Object|Array|Boolean|Number} value The value to be validated
 55              *  @returns Boolean true or false
 56              *  @type Boolean
 57              *  @example
 58              *      var v = new Validator();
 59              *      v.add('minLength', 5); 
 60              *      v.validate( myArray ); // returns true / false
 61              */
 62             validate: function(value) {
 63                 return _(myVs).all(function(valids, name) {
 64                     return _(valids).all(function(args) {
 65                         var allArgs = [];
 66                         allArgs.push(value);
 67                         allArgs = allArgs.concat(args);
 68                         return Validator[name].apply(this, allArgs);
 69                     });
 70                 });
 71             }
 72         };
 73     };

 75     /**
 76      *  Returns true if an array only contains unique members, false otherwise.
 77      *  @function
 78      *  @name Validator.unique
 79      *  
 80      *  @param {Array} arr The array to test
 81      *  @return {Boolean} True if all items are unique, false otherwise.
 82      */
 83     Validator.unique = function (arr) {
 84         return ( arr.length === _(arr).unique().length );
 85     };
 86     /**
 87      *  Returns true if an array is greater than or equal to a given length
 88      *  @function
 89      *  @name Validator.minLength
 90      *  
 91      *  @param {Array} arr The array to test
 92      *  @param {Number} min The minimum length
 93      *  @return {Boolean} True if the array is long enough, false otherwise.
 94      */
 95     Validator.minLength = function (arr, min) {
 96         return arr.length >= min;
 97     };
 98     /**
 99      *  Returns true if an array is less than or equal to a given length
100      *  @function
101      *  @name Validator.maxLength
102      *  
103      *  @param {Array} arr The array to test
104      *  @param {Number} min The maximum length
105      *  @return {Boolean} True if the array is short enough, false otherwise.
106      */
107     Validator.maxLength = function (arr, max) {
108         return arr.length <= max;
109     };
110     /**
111      *  Returns true if an array is within a size range, inclusive.
112      *  @function
113      *  @name Validator.lengthInRange
114      *  
115      *  @param {Array} arr The array to test
116      *  @param {Number} min The minimum length
117      *  @param {Number} max The maximum length
118      *  @return {Boolean} True if the array is neither too long nor too short, false otherwise.
119      */
120     Validator.lengthInRange = function (arr, min, max) {
121         return ( minLength(arr, min) && maxLength(arr, max) );
122     };
123     /**
124      *  Returns true if the argument matches a regular expression
125      *  @function
126      *  @name Validator.matchesRegex
127      *  
128      *  @param {String|String[]} strOrArr The string or array of strings to test
129      *  @param {RegExp} regex The regular expression to use for the test
130      *  @return {Boolean} True if all the strings match the regex
131      */
132     Validator.matchesRegex = function (strOrArr, regex) {
133         var arr = (typeof strOrArr === 'array') ? strOrArr : [strOrArr];
134         return _(arr).all(function(str) {
135             return str.match(regex);
136         });
137     };
138     /**
139      *  Very loose regex to match valid emails - generally prefers false positives to false negatives
140      *  @function
141      *  @name Validator.isEmail
142      *  
143      *  @param {String|String[]} strOrArr The string or array of strings to test
144      *  @return {Boolean} True if all the strings are valid emails
145      */
146     Validator.isEmail = function (strOrArr) {
147         return Validator.matchesRegex(strOrArr, /[^@]+@[a-z-]+\.[a-z\-\.]+[^\.]/i);
148     };

160     /**
161      *  Returns true if an html fragment has any textual or image content (i.e. isn't just empty <span> tags)
162      *  @function
163      *  @name Validator.htmlHasContent
164      *  
165      *  @param {String|String[]} strOrArr The string or array of strings to test
166      *  @return {Boolean} True if there is some text content or an image, false otherwise.
167      */
168     Validator.htmlHasContent = function (strOrArr) {
169         var arr = (typeof strOrArr !== 'string') ? strOrArr : [strOrArr],
170             isBrowser = (window && document && document.createElement),
171             div;
172         if ( !strOrArr ) return false;
173         if  ( !isBrowser ) {
174             throw new Error('Html support only available in a browser!');
175         }
176         div = document.createElement('div');
177         return _(arr).all(function(str) {
178             var cleaned, trimmed;
179             str = Validator.trim(str);
180             if ( str.length === 0 ) return false;
181             div.innerHTML = str;
182             cleaned = div.innerText || div.textContent;
183             trimmed = Validator.trim(cleaned);
184             return (trimmed.length > 0);
185         });
186     };
187     /**
188      *  Utility function to trim strings
189      *  @function
190      *  @name Validator.trim
191      *  
192      *  @param {String} str The string to trim
193      *  @return {String} The trimmed string
194      */
195     Validator.trim = function(str) {
196         return str.replace(/^\s+/, '').replace(/\s+$/, '');
197     };
